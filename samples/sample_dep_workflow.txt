name: Notify package manager in <REPOSITORY NAME>

on: 
  release:
    types: [published]

permissions:
  contents: read

jobs: 
  Notify:
    runs-on: ubuntu-latest
    env:
      GH_REPOSITORY_PAT: ${{ secrets.<PAT NAME> }}

    steps:
      - name: Checkout Repository
        uses: actions/checkout@v4

      - name: Authenticate
        run: |
          echo "$GH_REPOSITORY_PAT" | gh auth login --with-token

      - name: Notify central dependency repo
        run: |
          gh api repos/Team-Deepiri/deepiri-pkg-version-manager/dispatches \
            -f event_type="dependency_update" \
            -f client_payload="$(jq -n \
              --arg repo "${{ github.event.repository.name }}" \
              --arg sha "${{ github.sha }}" \
              --arg tag "${{ github.event.release.tag_name }}" \
              '{repo: $repo, sha: $sha, tag: $tag}')"

Implementation Notes:
 - Using the gh api across repos requires github authentication, so each repo that has a dispatch workflow needs an associated GH_REPOSITORY_PAT
   - Store the PAT in the repo secrets
   - Needs repo and workflow permissions (with read/write)
 - Replace <REPOSITORY NAME> with the name of the repo
 - Replace <PAT NAME> with the name you assign to the PAT
